history = model.fit(
    x={
        'seq_input':      X_train_seq,
        'hour_input':     hour_train,
        'dow_input':      dow_train,
        'days_gap_input': days_train.reshape(-1, 1)   # shape (N, 1)
    },
    y=y_train,
    validation_data=(
        {
            'seq_input':      X_val_seq,
            'hour_input':     hour_val,
            'dow_input':      dow_val,
            'days_gap_input': days_val.reshape(-1, 1)
        },
        y_val
    ),
    epochs=EPOCHS,
    batch_size=BATCH_SIZE,
    callbacks=callbacks,
    verbose=1
)

print(f"\nTraining complete. Best val_loss epoch saved to lstm_time_model.h5")
